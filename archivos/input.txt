int navidad() {

    //comentario
    int a;
    int b = 5;
    int numeros[5];

    a = 2;
    b += 3;
    numeros[0] = a + b;

    if (a < b) {
        a++;
    } else {
        a = a - 1;
    }

    for (int i = 0; i < 5; i++) {
        numeros[i] = i * 2;
    }

    while (a < 10) {
        a = a + 1;
    }

    do {
        a--;
    } while (a > 0);

    return a;
}
